 11   if (el.addEventListener) {
 12     el.addEventListener(eventName, handler);
 13   } else if (el.attachEvent) {
 14     el.attachEvent('on' + eventName, function(){
 15       handler.call(el);
 16     });
 17   }

 32 /**
 33   * @name ExcelPlus()
 34   * @constructor

 36   *
 37   * @param {Object} params
 38   *   @param {Boolean} [params.showErrors=true] The errors will be shown and will throw an error

 41 function ExcelPlus(params) {
 42   this.aLetters = [];
 43   this.aLetters["A"]=1, this.aLetters["B"]=2, this.aLetters["C"]=3, this.aLetters["D"]=4, this.aLetters["E"]=5, this.aLetters["F"]=6, this.aLetters["G"]=7, this.aLetters["H"]=8, this.aLetters["I"]=9, this.aLetters["J"]=10, this.aLetters["K"]=11, this.aLetters["L"]=12, this.aLetters["M"]=13, this.aLetters["N"]=14, this.aLetters["O"]=15, this.aLetters["P"]=16, this.aLetters["Q"]=17;
 44   this.aLetters["R"]=18, this.aLetters["S"]=19, this.aLetters["T"]=20, this.aLetters["U"]=21, this.aLetters["V"]=22, this.aLetters["W"]=23, this.aLetters["X"]=24, this.aLetters["Y"]=25, this.aLetters["Z"]=26;
 45   
 46   this.oSheet = null; // the current sheet that we are working on
 47   this.selectedSheet = ""; // name of the selected sheet
 48   this.nbSheets = 0; // number of sheets
 49   this.oFile = null; // the File object
 50   this.nbRows = 0, this.nbColumns = 0; // the number of rows and columns used in the opened file
 51   this.filename = ""; // name of the file we read
 52   this.usedRange = ""; // e.g. A1-C250
 53 
 54   params = params || {};
 55   this.error = "";
 56 
 57   // to show the errors if any, otherwise you can handle the errors using YourObject.errors
 58   this.showErrors = (params.showErrors === undefined ? true : params.showErrors);

151   /**
152    * @name ExcelPlus().openRemote
153    * @function
154    * @description This function read a remote Excel file
155    *
156    * @param {String} url The url of the Excel file (make sure it follows the cross domain security)
157    * @param {Function} onReady=function(passed){} This function is called when the Excel file has been read, and 'passed' will return true (success) or false (failed)
158    */
159   openRemote:function(url, onReady) {
160     var _this=this;
161     // get the remote file binary content
162     var xhr = new XMLHttpRequest();
163     var convertResponseBodyToText = function(e) { return e };
164     xhr.open("GET", url, true);
165     if (xhr.overrideMimeType) xhr.overrideMimeType("text/plain; charset=x-user-defined")
166     else {
167       // for old browsers like IE8
168       xhr.setRequestHeader("Accept-Charset", "x-user-defined");
169       // see http://stackoverflow.com/questions/1919972/how-do-i-access-xhr-responsebody-for-binary-data-from-javascript-in-ie
170       var VB_Fix_IE = '<script language="VBScript">'+"\r\n"
171                     + "Function IEBinaryToArray_ByteStr(Binary)\r\n"
172                     + "  IEBinaryToArray_ByteStr = CStr(Binary)\r\n"
173                     + "End Function\r\n"
174                     + "Function IEBinaryToArray_ByteStr_Last(Binary)\r\n"
175                     + "  Dim lastIndex\r\n"
176                     + "  lastIndex = LenB(Binary)\r\n"
177                     + "  if lastIndex mod 2 Then\r\n"
178                     + "    IEBinaryToArray_ByteStr_Last = Chr( AscB( MidB( Binary, lastIndex, 1 ) ) )\r\n"
179                     + "  Else\r\n"
180                     + "    IEBinaryToArray_ByteStr_Last = "+'""'+"\r\n"
181                     + "  End If\r\n"
182                     + "End Function\r\n"
183                     + "\<\/script>\r\n";
184       document.write(VB_Fix_IE);
185       
186       convertResponseBodyToText = function(binary) {
187         var byteMapping = {};
188         for ( var i = 0; i < 256; i++ ) {
189           for ( var j = 0; j < 256; j++ ) {
190             byteMapping[ String.fromCharCode( i + j * 256 ) ] = String.fromCharCode(i) + String.fromCharCode(j);
191           }
192         }
193         var rawBytes = IEBinaryToArray_ByteStr(binary);
194         var lastChr = IEBinaryToArray_ByteStr_Last(binary);
195         return rawBytes.replace(/[\s\S]/g, function( match ) { return byteMapping[match]; }) + lastChr;
196       };
197     }
198     xhr.onreadystatechange = function(event) {
199       if (xhr.readyState == 4) {
200         if(xhr.status == 200) {
201           _this.filename = url.split("/").slice(-1);
202           var data = (xhr.overrideMimeType ? xhr.responseText : convertResponseBodyToText(xhr.responseBody));
203           _this.oFile = XLSX.read(data, {type: 'binary'});
204           _this.nbSheets = _this.oFile.SheetNames.length;
205           if (_this.nbSheets == 1) {
206             _this.selectSheet(1);
207           }
208           onReady.call(_this, true);
209         } else {
210           _this.error = "[openRemote] Cannot find the remote file";
211           _this._showErrors();
212           onReady.call(_this, false)
213         }
214       }
215     };
216     xhr.send(null);
217     
218     return this;

236   /**
237    * @name ExcelPlus().selectSheet
238    * @function
239    * @description This function selects the specific sheet
240    *
241    * @param {String|Number} sheet If it's a number (starting from) that reprents the sheet position in the Excel file, otherwise it must be the name of the sheet (case sensitive)
242    */
243   selectSheet:function(sheet) {
244     if (this.oFile == null) {
245       this.error = "[selectSheet] No Excel file opened.";
246       this._showErrors();
247       return this;
248     }
249     
250     // if we have a number, then we search the related sheetname
251     if (!isNaN(sheet)) {
252       // find the sheet name based on the number
253       var arr = this.getSheetNames();
254       sheet-=1;
255       var len = arr.length;
256       if (sheet >= len) {
257         this.error = "[selectSheet] Impossible to select sheet #"+sheet+" because only "+len+" sheet"+(len>1?"s":"")+" found in the file.";
258         this._showErrors();
259         return this;
260       }
261       
262       for (var i=0; i < len; i++) {
263         if (sheet == i) {
264           sheet = arr[i];
265           break;
266         }
267       }
268     }
269     if (!isNaN(sheet)) {
270       this.error = "[selectSheet] Impossible to select sheet #"+sheet+" because I cannot find it.";
271       this._showErrors();
272       return this;
273     }
274 
275     // make sure the provided name exists
276     if (this.oFile.SheetNames.indexOf(sheet) === -1) {
277       this.error = "[selectSheet] Impossible to select sheet "+sheet+" because there is no sheet with that name.";
278       this._showErrors();
279       return this;
280     }
281     
282     this.oSheet      = this.oFile.Sheets[sheet];
283     this.selectedSheet = sheet;
284     if (this.oSheet["!ref"]) {
285       this.usedRange = this.oSheet["!ref"]; // e.g. A1-C250
286       var lastCell   = this._getCellCoord(this.usedRange.split(":")[1]);
287       this.nbRows    = lastCell.row * 1;
288       this.nbColumns = lastCell.column * 1;
289     }
290     
291     return this;
292   },
293   /**
294    * @name ExcelPlus().readAll
295    * @function
296    * @description This function will read the all of the used range (we assume that the content starts at A1)
297    *
298    * @return {Array} it returns a 2D array; the first one is the rows, and the second one is the columns
299    */
300   readAll:function() {
301     if (this._checkSheet()==false) {
302       this.error = "[readAll] You have to select a sheet before using `readAll`";
303       this._showErrors();
304       return this;
305     }
306     
307     var table = [],line,val,cell;
308     for (var row=1; row <= this.nbRows; row++) {
309       line = [];
310       for (var col=1; col <= this.nbColumns; col++) {
311         val="";
312         cell=this._getColumnName(col) + "" + row;
313         if (this.oSheet[cell]) val = this.oSheet[cell].v;
314         line.push(val);
315       }
316       table.push(line);
317     }
318     return table;
319   },
320   /**
321    * @name ExcelPlus().read
322    * @function
323    * @description This function permits to read a cell or a range
324    *
325    * @param {String} range can be a range (e.g. "A1:D1"), or a single cell (e.g. "A1")
326    * @return {Array} if it's a range then it returns a 2D array (the first one is the rows, and the second one is the columns), if it's a cell then it returns the value of the cell
327    */
328   read:function(range) {
329     if (this._checkSheet()==false) return this
330 
331     if (this._isRange(range)) {
332       range = XLSX.utils.decode_range(range);
333       var table=[], line=[], R, C;
334       for (R = range.s.r; R <= range.e.r; ++R) {
335         line = [];
336         for (C = range.s.c; C <= range.e.c; ++C) {
337           val = this.oSheet[XLSX.utils.encode_cell({c:C, r:R})];
338           if (val) val = val.v;
339           else val = null;
340           line.push(val);
341         }
342         table.push(line);
343       }
344       return table;
345     }
346     else if (this._isCell(range)) {
347       return this.oSheet[range].v || "";
348     }
349     
350     return null;
351   },

375   /**
376    * @name ExcelPlus().write
377    * @function
378    * @description Write values into the workbook
379    *
380    * @param {Object} options
381    *   @param {String} [sheet] You can give the name of the sheet where to write, or the current selected sheet will be use, or the first sheet will be used
382    *   @param {String} [cell] You can specify the cell reference (e.g. "A1")
383    *   @param {String|Boolean|Number|Date|Array} content If it's an arrya then we ignore the "cell" param (e.g. [ [ "Content Cell A1", "Content Cell B1" ] [ "Content Cell A2", "Content Cell B2" ] ], otherwise the "cell" needs to be provided
384    *
385    * @example
386    *   var my=new ExcelPlus();
387    *   my.createFile(["Book1","Book2"]);
388    *   my.write({ "sheet":"Book1", "content":[ ["A1","B1","C1"] ] });
389    *   my.write({ "sheet":"Book2", "cell":"A1", "content":"This is the cell A1" });
390    *
391    * // TODO the value can also be a formula, e.g. "=SUM(A1:B1)"
392    */
393   write:function(options) {
394     options.sheet = options.sheet || this.selectedSheet || this.oFile.SheetNames[0];
395     // select the sheet requested here
396     this.selectSheet(options.sheet);
397     
398     if (!options.content) {
399       this.error = "[write] Please provide the 'content'";
400       this._showErrors();
401       return this;
402     }
403     // check if "content" is an array, then ignore "cell"
404     if (Array.isArray(options.content)) options.cell=null;
405     else {
406       if (!options.cell) {
407         this.error = "[write] Please provide the 'cell' reference";
408         this._showErrors();
409         return this;
410       }
411     }
412     
413     if (!options.cell) {
414       var data=options.content, R, C, cell_ref, cell;
415       for (R = 0; R != data.length; ++R) {
416         for (C = 0; C != data[R].length; ++C) {
417           cell = this._createCell(data[R][C]);
418           if (cell.v == null) continue;
419           cell_ref = XLSX.utils.encode_cell({c:C,r:R});
420           this.oFile.Sheets[this.selectedSheet][cell_ref] = cell;
421           this._updateUsedRange(cell_ref);
422         }
423       }
424     } else {
425       this.writeCell(options.cell, options.content);
426     }
427     
428     this.oSheet = this.oFile.Sheets[this.selectedSheet];
429     
430     return this;
431   },
432   /**
433    * @name ExcelPlus().writeCell
434    * @function
435    * @description Write a value into a cell for the current selected sheet
436    *
437    * @param {String} cell The cell reference (e.g. "A1")
438    * @param {String|Number|Date|Boolean} content This is the value for the cell (could be `null`)
439    *
440    * @example
441    *   var my=new ExcelPlus();
442    *   my.createFile(["Book1","Book2"]);
443    *   my.selectSheet("Book1");
444    *   my.writeCell("A1","This is the content of the first cell");
445    *
446    * // TODO the value can also be a formula, e.g. "=SUM(A1:B1)"
447    */
448   writeCell:function(cell, content) {
449     if (!this.selectedSheet) {
450       if (this.oFile.SheetNames.length>0) {
451         this.error = "[writeCell] No sheet selected";
452         this._showErrors();
453         return this;
454       } else {
455         this.error = "[writeCell] You have to create a sheet first";
456         this._showErrors();
457         return this;
458       }
459     }
460     
461     if (!content) {
462       this.error = "[writeCell] You have to provide the content";
463       this._showErrors();
464       return this;
465     }
466     if (!this._isCell(cell)) {
467       this.error = "[writeCell] You have to provide a cell reference, e.g. 'A1'";
468       this._showErrors();
469       return this;
470     }
471     
472     // try to select the sheet to make sure there is no error
473     if (content == null) delete this.oFile.Sheets[this.selectedSheet][cell];
474     else this.oFile.Sheets[this.selectedSheet][cell] = this._createCell(content);
475     this._updateUsedRange(cell);
476     // update the info about the selected sheet
477     this.selectSheet(this.selectedSheet);
478     
479     return this;
480   },

528   /**
529    * @name ExcelPlus().saveAs
530    * @function
531    * @description This function permits de save the current book to a real Excel file (only for Modern Browsers and IE10+)
532    *
533    * @param {String} filename The name of the file, .e.g "book1.xlsx"
534    */
535   saveAs:function(filename) {
536     filename = filename || "book1.xlsx";
537     var ext = filename.split(".").pop();
538     // check the type/extension
539     if (['xlsx', 'xlsm', 'xlsb'].indexOf(ext) === -1) {
540       ext = "xlsx";
541       filename += "."+ext;
542     }
543     
544     if (typeof Uint8Array === "undefined") {
545       this.error = "[saveAs] Sorry but this function is only supported by modern browsers";
546       this._showErrors();
547       return this;
548     }
549     
550     var wbout = XLSX.write(this.oFile, {bookType:ext, bookSST:true, type: 'binary'});
551     var s2ab = function(s) {
552       var buf = new ArrayBuffer(s.length);
553       var view = new Uint8Array(buf);
554       for (var i=0; i!=s.length; ++i) view[i] = s.charCodeAt(i) & 0xFF;
555       return buf;
556     }
557     saveAs(new Blob([s2ab(wbout)],{type:"application/octet-stream"}), filename);
558     
559     return this;
560   },

574   /**
575    * @ignore
576    * This function takes a value and converts it to a `cell` format defined by SheetJS
577    *
578    * @param {String|Boolean|Date|Number} value The value for a cell
579    * @return {Object} It will return a cell object with 't' for the format, 'v' for the formatted value, etc (see https://github.com/SheetJS/js-xlsx#cell-object)
580    */
581   _createCell:function(value) {
582     var cell = {"v":value, "t":this._getValueType(value)};
583     if (value instanceof Date) {
584       cell.z = XLSX.SSF._table[14];
585       cell.v = this._dateToNum(value);
586     }

590   /**
591    * @ignore
592    * This function permits to update the used range for a sheet
593    * It's based on the SheetJS example code
594    *
595    * @param {Object|String} coord It's the coordonate of a cell e.g. {column:1, row:1} or "A1"
596    */
597   _updateUsedRange:function(coord) {
598     var range = XLSX.utils.decode_range(this.oSheet["!ref"]);
599     if (this._isCell(coord)) coord=this._getCellCoord(coord);
600     coord.row-=1;
601     coord.column-=1;
602     var anyChange = false
